import axios from "axios";
import xml2js from "xml2js";
import db from "../db";

export async function syncNetwork(req, res) {
  const { network } = req.params;

  try {
    if (network === "admitad") {
      const feedUrl = process.env.ADMITAD_FEED_URL;
      const xml = await axios.get(feedUrl);
      const parsed = await xml2js.parseStringPromise(xml.data);

      const items = parsed?.yml_catalog?.shop?.[0]?.offers?.[0]?.offer || [];

      for (const item of items.slice(0, 100)) {
        await db.query(
          `INSERT INTO affiliate_products
          (network, external_id, title, price, currency, image, affiliate_url)
          VALUES ($1,$2,$3,$4,$5,$6,$7)
          ON CONFLICT DO NOTHING`,
          [
            "admitad",
            item.$.id,
            item.name?.[0],
            item.price?.[0],
            "USD",
            item.picture?.[0],
            item.url?.[0]
          ]
        );
      }
    }

    if (network === "aliexpress") {
      // placeholder for Portals API
      throw new Error("AliExpress API not yet configured");
    }

    res.json({ success: true });
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Sync failed", details: err.message });
  }
}
