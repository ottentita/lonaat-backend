🔧 STEP 1 — CREATE ROUTE FILE (MANDATORY)
Create file:

/server/routes/admitad.routes.js
Paste this EXACT code:
Copy code
Js
const express = require("express");
const axios = require("axios");
const xml2js = require("xml2js");

const router = express.Router();

router.get("/search", async (req, res) => {
  try {
    const query = (req.query.q || "").toLowerCase();

    if (!query) {
      return res.status(400).json({ error: "Missing search query" });
    }

    const feedUrl = process.env.ADMITAD_FEED_URL;

    if (!feedUrl) {
      return res.status(500).json({ error: "ADMITAD_FEED_URL not set" });
    }

    const response = await axios.get(feedUrl);
    const parsed = await xml2js.parseStringPromise(response.data, {
      explicitArray: false,
    });

    const items = parsed?.yml_catalog?.shop?.offers?.offer || [];

    const products = Array.isArray(items) ? items : [items];

    const results = products.filter(p =>
      (p.name && p.name.toLowerCase().includes(query)) ||
      (p.categoryId && p.categoryId.toLowerCase().includes(query))
    ).slice(0, 50);

    return res.json({
      success: true,
      count: results.length,
      data: results
    });

  } catch (err) {
    console.error("Admitad search error:", err.message);
    res.status(500).json({ error: "Failed to fetch Admitad products" });
  }
});

module.exports = router;
