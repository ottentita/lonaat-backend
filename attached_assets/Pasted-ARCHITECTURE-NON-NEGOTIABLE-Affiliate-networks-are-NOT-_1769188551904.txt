router.post("/:network/sync-products", async (req, res) => {
  const network = req.params.network;

  const config = AFFILIATE_NETWORKS[network];
  if (!config) {
    return res.status(404).json({ error: "Network not supported" });

    });
  }

  const products = await importProductsFromNetwork(network);
  res.json({ success: true, productsImported: products.length });
});

export async function importProductsFromNetwork(network: string) {
  const feed = await fetchFeed(network);
  const products = normalize(feed);

  for (const product of products) {
    await db.products.upsert({
      title: product.title,
      price: product.price,
      image: product.image,
      affiliate_url: product.url,
      source_network: network,
      status: "active"
    });
  }

  return products;
}
