Hash Generation Logic

import crypto from "crypto";

export function generateLandHash(land: any) {
  const data = `${land.title_number}|${land.owner_name}|${land.area_sqm}|${land.polygon}`;
  return crypto.createHash("sha256").update(data).digest("hex");
}
Database
Copy code
