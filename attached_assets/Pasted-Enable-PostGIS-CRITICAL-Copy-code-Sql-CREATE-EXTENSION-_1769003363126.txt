CREATE TABLE lands (
  id SERIAL PRIMARY KEY,

  title_number VARCHAR(100) UNIQUE NOT NULL,
  land_name VARCHAR(255),

  owner_id INTEGER REFERENCES users(id),
  owner_name VARCHAR(255) NOT NULL,

  region VARCHAR(100) NOT NULL,
  town VARCHAR(100),
  neighborhood VARCHAR(100),

  area_sqm NUMERIC(12,2),

  -- GPS DATA
  polygon GEOMETRY(POLYGON, 4326) NOT NULL,
  centroid GEOMETRY(POINT, 4326),

  status VARCHAR(30) DEFAULT 'active', -- active | sold | blocked
  is_locked BOOLEAN DEFAULT false,

  created_at TIMESTAMP DEFAULT NOW()
);

import db from "../db";

export async function checkDuplicateLand(polygonWKT: string) {
  const query = `
    SELECT id, title_number
    FROM lands
    WHERE ST_Intersects(
      polygon,
      ST_GeomFromText($1, 4326)
    )
  `;
  const result = await db.query(query, [polygonWKT]);
  return result.rows;
}

If rows.length > 0 → BLOCK SALE

4️⃣ Land Registration API

router.post("/register", auth, async (req, res) => {
  const {
    title_number,
    land_name,
    polygonWKT,
    region,
    town,
    neighborhood,
    area_sqm
  } = req.body;

  const duplicates = await checkDuplicateLand(polygonWKT);

  if (duplicates.length > 0) {
    return res.status(409).json({
      error: "Duplicate land detected",
      conflicts: duplicates
    });
  }

  await db.query(`
    INSERT INTO lands (
      title_number, land_name, owner_id, owner_name,
      region, town, neighborhood, area_sqm,
      polygon, centroid
    )
    VALUES (
      $1,$2,$3,$4,$5,$6,$7,$8,
      ST_GeomFromText($9,4326),
      ST_Centroid(ST_GeomFromText($9,4326))
    )
  `, [
    title_number,
    land_name,
    req.user.id,
    req.user.name,
    region,
    town,
    neighborhood,
    area_sqm,
    polygonWKT
  ]);

  res.json({ success: true });
});

router.get("/map", auth, async (_, res) => {
  const lands = await db.query(`
    SELECT
      id,
      land_name,
      title_number,
      owner_name,
      ST_AsGeoJSON(polygon) AS polygon,
      status
    FROM lands
  `);

  res.json(lands.rows);
});
