import axios from "axios";

export async function searchAffiliateOffers(
  network: string,
  query: string
) {
  switch (network.toLowerCase()) {
    case "digistore24":
      return searchDigistore24(query);
    case "awin":
      return searchAwin(query);
    case "mylead":
      return searchMyLead(query);
    default:
      throw new Error("Unsupported affiliate network");
  }
}

/* ---------------- DIGISTORE24 ---------------- */

async function searchDigistore24(query: string) {
  const vendorId = process.env.DIGISTORE24_VENDOR_ID;
  const apiKey = process.env.DIGISTORE24_API_KEY;

  if (!vendorId || !apiKey) {
    throw new Error("Digistore24 credentials missing");
  }

  const response = await axios.get(
    "https://api.digistore24.com/v1/products",
    {
      params: {
        vendorId,
        search: query
      },
      headers: {
        Authorization: `Bearer ${apiKey}`
      }
    }
  );

  return response.data.products || [];
}

/* ---------------- AWIN ---------------- */

async function searchAwin(query: string) {
  const token = process.env.AWIN_API_TOKEN;

  if (!token) {
    throw new Error("Awin API token missing");
  }

  const response = await axios.get(
    "https://api.awin.com/publishers/advertisers",
    {
      params: { q: query },
      headers: {
        Authorization: `Bearer ${token}`
      }
    }
  );

  return response.data || [];
}

/* ---------------- MYLEAD ---------------- */

async function searchMyLead(query: string) {
  const apiKey = process.env.MYLEAD_API_KEY;

  if (!apiKey) {
    throw new Error("MyLead API key missing");
  }

  const response = await axios.get(
    "https://api.mylead.global/v1/offers",
    {
      params: { search: query },
      headers: {
        "X-Api-Key": apiKey
      }
    }
  );

  return response.data.offers || [];
}

import { Router } from "express";
import { PrismaClient } from "@prisma/client";
import { searchAffiliateOffers } from "../services/affiliateSearch";

const router = Router();
const prisma = new PrismaClient();

router.get("/search", async (req, res) => {
  const { network, q } = req.query;

  if (!network || !q) {
    return res.status(400).json({ message: "network and query required" });
  }

  try {
    const offers = await searchAffiliateOffers(
      String(network),
      String(q)
    );

    res.json({ offers });
  } catch (err: any) {
    console.error(err.message);
    res.status(500).json({ message: err.message });
  }
});

router.post("/import", async (req, res) => {
  const { products } = req.body;

  if (!Array.isArray(products)) {
    return res.status(400).json({ message: "products array required" });
  }

  await prisma.product.createMany({
    data: products.map(p => ({
      name: p.name,
      description: p.description || "",
      price: Number(p.price) || 0,
      network: p.network,
      status: "ACTIVE",
      isPublished: true
    })),
    skipDuplicates: true
  });

  res.json({ message: "Products imported successfully" });
});

export default router;
