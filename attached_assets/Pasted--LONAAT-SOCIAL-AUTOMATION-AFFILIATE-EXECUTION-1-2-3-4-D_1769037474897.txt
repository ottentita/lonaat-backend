STEP 3 — SOCIAL QUEUE SERVICE
📁 backend/src/services/socialQueue.ts
Copy code
Ts
import { generateContent } from "./socialAI";
import db from "../db";

export async function enqueueSocialPosts(product) {
  const content = generateContent(product);

  for (const platform of Object.keys(content)) {
    await db.query(
      `INSERT INTO social_posts (product_id, platform, content)
       VALUES ($1,$2,$3)`,
      [product.id, platform, content[platform]]
    );
  }
}
