// 2. Search products
export async function searchProducts(req, res) {
  const { network, q } = req.query;

  if (!network || !q) {
    return res.status(400).json({ error: "Missing params" });
  }

  // TEMP: return preview results
  res.json({ products: [] });
}

// 3. Import product
export async function importProduct(req, res) {
  const p = req.body;

  await db.query(`
    INSERT INTO affiliate_products
    (network, external_id, title, price, currency, image, affiliate_url, category)
    VALUES ($1,$2,$3,$4,$5,$6,$7,$8)
  `, [
    p.network,
    p.external_id,
    p.title,
    p.price,
    p.currency,
    p.image,
    p.affiliate_url,
    p.category
  ]);

  res.json({ success: true });
}

// 4. Bulk import
export async function bulkImport(req, res) {
  const { products } = req.body;

  for (const p of products) {
    await db.query(`
      INSERT INTO affiliate_products
      (network, external_id, title, price, currency, image, affiliate_url, category)
      VALUES ($1,$2,$3,$4,$5,$6,$7,$8)
    `, [
      p.network,
      p.external_id,
      p.title,
      p.price,
      p.currency,
      p.image,
      p.affiliate_url,
      p.category
    ]);
  }

  res.json({ success: true });
}

// 5. Click tracking
export async function trackClick(req, res) {
  const productId = req.params.id;
  const userId = req.query.user;

  const product = await db.query(
    "SELECT affiliate_url FROM affiliate_products WHERE id=$1",
    [productId]
  );

  await db.query(`
    INSERT INTO affiliate_clicks (product_id, user_id)
    VALUES ($1,$2)
  `, [productId, userId]);

  res.redirect(product.rows[0].affiliate_url);
}
