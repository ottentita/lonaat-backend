import db from "../db";

export async function detectDuplicate(polygonWKT: string) {
  const query = `
    SELECT id, title_number
    FROM lands
    WHERE ST_Intersects(
      polygon,
      ST_GeomFromText($1, 4326)
    )
  `;
  const result = await db.query(query, [polygonWKT]);
  return result.rows;
}

import express from "express";
import { detectDuplicate } from "../services/landValidation";
import auth from "../middleware/auth";
import db from "../db";

const router = express.Router();

/* REGISTER LAND */
router.post("/register", auth, async (req, res) => {
  const { title_number, land_name, polygonWKT, region, town, area_sqm } = req.body;

  const duplicates = await detectDuplicate(polygonWKT);

  if (duplicates.length > 0) {
    return res.status(409).json({
      error: "Duplicate land detected",
      conflicts: duplicates
    });
  }

  await db.query(`
    INSERT INTO lands (
      title_number, land_name, owner_id, owner_name,
      region, town, area_sqm,
      polygon, centroid
    )
    VALUES (
      $1,$2,$3,$4,$5,$6,$7,
      ST_GeomFromText($8,4326),
      ST_Centroid(ST_GeomFromText($8,4326))
    )
  `, [
    title_number,
    land_name,
    req.user.id,
    req.user.name,
    region,
    town,
    area_sqm,
    polygonWKT
  ]);

  res.json({ success: true });
});

/* MAP DATA */
router.get("/map", auth, async (_, res) => {
  const lands = await db.query(`
    SELECT
      id,
      land_name,
      title_number,
      owner_name,
      status,
      ST_AsGeoJSON(polygon) AS polygon
    FROM lands
  `);
  res.json(lands.rows);
});

export default router;
