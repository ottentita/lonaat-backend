import axios from "axios";
import { db } from "../db";

export async function syncAdmitad(req, res) {
  try {
    const feedUrl = process.env.ADMITAD_FEED_URL;
    const response = await axios.get(feedUrl);

    const products = parseAdmitadXML(response.data);

    for (const p of products) {
      await db.product.upsert({
        where: { externalId: p.id },
        update: p,
        create: {
          ...p,
          source: "ADMITAD",
          published: true
        }
      });
    }

    res.json({ success: true, count: products.length });
  } catch (e) {
    res.status(500).json({ error: "Admitad sync failed" });
  }
}

📁 syncAliExpress
Copy code
Ts
export async function syncAliExpress(req, res) {
  const apiKey = process.env.ALIEXPRESS_API_KEY;

  const response = await axios.get(
    "https://api.aliexpress.com/product/list",
    { headers: { Authorization: apiKey } }
  );

  for (const p of response.data.products) {
    await db.product.upsert({
      where: { externalId: p.id },
      update: p,
      create: {
        name: p.title,
        price: p.price,
        image: p.image,
        url: p.affiliate_link,
        source: "ALIEXPRESS",
        published: true
      }
    });
  }

  res.json({ success: true });
}
